Add-Type -AssemblyName System.Web

function Write-JsonResponse {
    param(
        [int]$StatusCode,
        [hashtable]$Payload
    )

    $reason = switch ($StatusCode) {
        200 { 'OK' }
        400 { 'Bad Request' }
        405 { 'Method Not Allowed' }
        500 { 'Internal Server Error' }
        default { '' }
    }

    [Console]::OutputEncoding = [System.Text.Encoding]::UTF8
    Write-Output "Status: $StatusCode $reason"
    Write-Output "Content-Type: application/json; charset=utf-8"
    Write-Output ""
    Write-Output ($Payload | ConvertTo-Json -Depth 10 -Compress)
    exit
}

try {
    if (($env:REQUEST_METHOD ?? '') -ne 'POST') {
        Write-JsonResponse -StatusCode 405 -Payload @{
            ok = $false
            error = 'Uzyj metody POST.'
        }
    }

    $rawBody = [Console]::In.ReadToEnd()
    if ([string]::IsNullOrWhiteSpace($rawBody)) {
        Write-JsonResponse -StatusCode 400 -Payload @{
            ok = $false
            error = 'Brak danych do zapisania.'
        }
    }

    $parsedForm = [System.Web.HttpUtility]::ParseQueryString($rawBody)
    $payloadJson = $parsedForm['payload']
    if ([string]::IsNullOrWhiteSpace($payloadJson)) {
        $payloadJson = $rawBody
    }

    $payload = $payloadJson | ConvertFrom-Json
    if ($null -eq $payload -or $null -eq $payload.files -or $payload.files.Count -eq 0) {
        Write-JsonResponse -StatusCode 400 -Payload @{
            ok = $false
            error = 'Brak plikow do zapisania.'
        }
    }

    $exportDate = [string]$payload.exportDate
    if ($exportDate -notmatch '^\d{4}-\d{2}-\d{2}$') {
        Write-JsonResponse -StatusCode 400 -Payload @{
            ok = $false
            error = 'Niepoprawny format daty. Oczekiwano yyyy-MM-dd.'
        }
    }

    $rootDir = 'D:\PROD_REPO_DATA\AutomateDeploy\Deploys'
    $targetDir = Join-Path $rootDir $exportDate
    if (-not (Test-Path -LiteralPath $targetDir)) {
        New-Item -Path $targetDir -ItemType Directory -Force | Out-Null
    }

    $savedFiles = New-Object System.Collections.Generic.List[string]
    foreach ($file in $payload.files) {
        $fileName = [System.IO.Path]::GetFileName(([string]$file.filename).Trim())
        if ([string]::IsNullOrWhiteSpace($fileName)) {
            continue
        }

        if (-not $fileName.EndsWith('.json', [System.StringComparison]::OrdinalIgnoreCase)) {
            $fileName += '.json'
        }

        $targetFile = Join-Path $targetDir $fileName
        [System.IO.File]::WriteAllText($targetFile, ([string]$file.content), (New-Object System.Text.UTF8Encoding($false)))
        $savedFiles.Add($fileName) | Out-Null
    }

    if ($savedFiles.Count -eq 0) {
        Write-JsonResponse -StatusCode 400 -Payload @{
            ok = $false
            error = 'Nie znaleziono poprawnych nazw plikow do zapisania.'
        }
    }

    Write-JsonResponse -StatusCode 200 -Payload @{
        ok = $true
        directory = $targetDir
        saved = $savedFiles.Count
        files = $savedFiles
        server = [string]$payload.server
    }
}
catch {
    Write-JsonResponse -StatusCode 500 -Payload @{
        ok = $false
        error = $_.Exception.Message
    }
}
